import cv2
import numpy as np
from matplotlib import pyplot as plt


def linear_smoothing(image, kernel_size):
    return cv2.blur(image, (kernel_size, kernel_size))


def nonlinear_smoothing(image, kernel_size):
    return cv2.medianBlur(image, kernel_size)


# Read the image
image = cv2.imread("th.jpg", cv2.IMREAD_GRAYSCALE)
# Apply linear smoothing
linear_smoothed_image = linear_smoothing(image, kernel_size=5)
# Apply nonlinear smoothing
nonlinear_smoothed_image = nonlinear_smoothing(image, kernel_size=5)

# Display the original and smoothed images
plt.figure(figsize=(10, 5))
plt.subplot(1, 3, 1)
plt.imshow(image, cmap="gray")
plt.title("Original Image")
plt.axis("off")

plt.subplot(1, 3, 2)
plt.imshow(linear_smoothed_image, cmap="gray")
plt.title("Linear Smoothing")
plt.axis("off")

plt.subplot(1, 3, 3)
plt.imshow(nonlinear_smoothed_image, cmap="gray")
plt.title("Nonlinear Smoothing")
plt.axis("off")
plt.show()
